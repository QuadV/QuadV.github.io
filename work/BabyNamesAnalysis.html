years = range(1880, 2011)

pieces = []
columns = ['name', 'sex', 'births']

for year in years:
    path = '.\\pydata-book-master\\ch02\\names\\yob%d.txt' %year
    frame = pd.read_csv(path, sep=',', names=columns)
    frame['year'] = year
    pieces.append(frame)
names = pd.concat(pieces, ignore_index=True)

Add a new column 'prop' to table by grouping by year and sex. This column contains the birth name ratio to the whole set¶

In [6]:
def add_group(group):
    births = group.births.astype(float)
    group['prop'] = births/births.sum()
    return group

names = names.groupby(['year', 'sex']).apply(add_group)
